## 解法1 复杂度 O(n^2)
两次遍历的方法去穷举
```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function (nums, target) {
    let res = [];
    let num;
    for (let i = 0; i < nums.length; i++) {
        for (let j = 0; j < nums.length; j++) {
            if (i != j) {
                num = nums[i] + nums[j];
                if (num === target) {
                    res = [i, j];
                }
            }
        }
    }
    return res;
};
```
