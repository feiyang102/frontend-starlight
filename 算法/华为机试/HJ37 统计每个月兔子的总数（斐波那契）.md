// 斐波那契通项公式求解
var getCount = function (n) {
    let k = Math.sqrt(5);
    let a = ((1 + k) / 2) ** n;
    let b = ((1 - k) / 2) ** n;
    return Math.floor((1 / k) * (a - b));
};
